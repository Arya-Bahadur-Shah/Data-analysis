        'Mean': numeric_columns.mean(),
        'Standard Deviation': numeric_columns.std(),

        'Kurtosis': numeric_columns.kurt()

from scipy.stats import f_oneway
required_data = dataset[['Complaint Type', 'Request_Closing_Time']]
unique_complaints = required_data['Complaint Type'].unique()
groups = []
for complaint_types in unique_complaints:
    matched_rows = required_data['Complaint Type'] == complaint_types
    complaint_data = required_data[matched_rows]
    closing_times = complaint_data ['Request_Closing_Time'].dt.total_seconds()
    groups.append(closing_times)

f_stat, p_val = f_oneway(*groups)
print("F-Statistic:", f_stat)
print("P-Value:", p_val)
if p_val < 0.05:
    print("Reject Null Hypothesis: There is a difference in the average response time across complaint types.")
else:
    print("Fail to Reject Null Hypothesis: There is not difference in average response time accross complaint types.")
